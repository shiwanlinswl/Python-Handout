2. 网卡IP配置

网卡配置文件位置：/etc/sysconfig/network-scripts

[root@itcast ~]# ls /etc/sysconfig/network-scripts
ifcfg-ens33  ifdown-eth   ifdown-post    ifdown-Team      ifup-aliases  ifup-ipv6   ifup-post    ifup-Team      init.ipv6-global
ifcfg-lo     ifdown-ippp  ifdown-ppp     ifdown-TeamPort  ifup-bnep     ifup-isdn   ifup-ppp     ifup-TeamPort  network-functions
ifdown       ifdown-ipv6  ifdown-routes  ifdown-tunnel    ifup-eth      ifup-plip   ifup-routes  ifup-tunnel    network-functions-ipv6
ifdown-bnep  ifdown-isdn  ifdown-sit     ifup             ifup-ippp     ifup-plusb  ifup-sit     ifup-wireless

在目录中网卡的配置文件命名格式：ifcfg-网卡名称

[root@itcast ~]# cat /etc/sysconfig/network-scripts/ifcfg-ens33
TYPE=Ethernet
PROXY_METHOD=none
BROWSER_ONLY=no
BOOTPROTO=dhcp　　　　　　# BOOTPROTO的值可以设置为dhcp或static，dhcp：dhcp服务器获取IP，static：手工配置IP
DEFROUTE=yes
IPV4_FAILURE_FATAL=no
IPV6INIT=yes
IPV6_AUTOCONF=yes
IPV6_DEFROUTE=yes
IPV6_FAILURE_FATAL=no
IPV6_ADDR_GEN_MODE=stable-privacy
NAME=ens33
UUID=781e7103-b4a8-46c5-9344-26ae94cf6292
DEVICE=ens33
ONBOOT=no  　# ONBOOT可以设置为yes或no，yes：重启网络服务生效此配置，no：重启网络服务不生效此配置

例1：将ens33网卡IP固定为192.168.80.128

[root@itcast ~]# cat /etc/sysconfig/network-scripts/ifcfg-ens33
TYPE=Ethernet
PROXY_METHOD=none
BROWSER_ONLY=no
BOOTPROTO=static   # IP通过手工配置
IPADDR=192.168.80.128 # 设置IP地址
NETMASK=255.255.255.0 # 设置子网掩码
GATEWAY=192.168.80.2  # 设置网关
DEFROUTE=yes
IPV4_FAILURE_FATAL=no
IPV6INIT=yes
IPV6_AUTOCONF=yes
IPV6_DEFROUTE=yes
IPV6_FAILURE_FATAL=no
IPV6_ADDR_GEN_MODE=stable-privacy
NAME=ens33
UUID=781e7103-b4a8-46c5-9344-26ae94cf6292
DEVICE=ens33
ONBOOT=yes    # 设置重启网络服务生效此配置

3. 重启网络服务

修改完网卡配置之后，需要重启网络服务才能够生效或者重启服务器

# 重启网络服务命令
[root@itcast ~]# systemctl restart network

注意：在CentOS 7中服务管理用的是systemctl命令，在CentOS 6中服务管理用的是service命令

# systemctl命令用法
systemctl status|restart|start|stop 服务名.service

# service命令用法
service 服务名 start|stop|restart|status

start：启动

stop：停止

status：状态

restart：重启
