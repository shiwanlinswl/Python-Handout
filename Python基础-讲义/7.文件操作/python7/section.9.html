<2>参考代码

import os

path = input("请输入需要批量修改文件名的文件夹名称：")
# 跳转路径
os.chdir(path)
# # 获取当前的绝对路径
# full_path = os.getcwd()
# print(full_path)
# 遍历目标文件夹
for subpath_name in os.listdir():
    new_file_name = "【复件】" + subpath_name
    # 每个文件改名
    os.rename(subpath_name, new_file_name)
