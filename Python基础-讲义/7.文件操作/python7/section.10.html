def save_data():
    """保存数据"""
    print("功能：保存数据")
    # 打开文件
    f = open("cards.data", "w")
    # 写入数据
    f.write(str(g_card_list))
    # 关闭文件
    f.close()
    print("保存数据成功")


def load_data():
    """读取数据"""
    # 判断文件是否存在
    if os.path.exists("cards.data"):
        # 打开文件
        f = open("cards.data", "r")
        global g_card_list
        # 读取内容  如果想要将列表/元组格式的字符串转回列表/元组的类型，需要使用eval进行转换
        g_card_list = eval(f.read())
        # 关闭文件
        f.close()
